一.动手写ping扫描

很多机器都禁用ping扫描了,但是我刚刚入门,还是要学的.
ping.py

import logging
logging.getLogger("scapy.runtime").setLevel(logging.ERROR)
# 导入scapy报错日志,清除掉报错日志
from scapy.layers.inet import IP, ICMP
from scapy.all import *
from random import randint


def scapy_ping_one(host):
    id_ip = randint(1, 65535)  # 随机整数产生器,随机产生ip的id (随机才能伪装)
    id_ping = randint(1, 65535)  # 随机产生icmp的id
    seq_ping = randint(1, 65535) # ICMP序列号
    packet = IP(dst=host, ttl=1, id=id_ip)/ICMP(id=id_ping, seq=seq_ping)/b'welcome to wuhan'
    ping = sr1(packet, timeout=2, verbose=False)
    ping.show()
    if ping:  # 如果有结果就退出
        os._exit(3)  # 自定义退出码为3,区分多进程


if __name__ == '__main__':
    scapy_ping_one("127.0.0.1")
# dst 目的IP地址
# ttl自连为1
# IP和ICMP的id位为随机产生

ping1.py

import logging
logging.getLogger("scapy.runtime").setLevel(logging.ERROR)

from scapy.layers.inet import IP, ICMP

import ipaddress  # 可以写入地址段,且为数组 192.168.1.0/24
import time
import multiprocessing
from ping import scapy_ping_one # 从ping.py里面导入函数scapy_ping_one
from scapy.all import *
def scapy_ping_scan(network):
    net = ipaddress.ip_network(network)
    ip_processes = {}
    for ip in net:
        ip_addr = str(ip)
        ping_one = multiprocessing.Process(target=scapy_ping_one, args=(ip_addr,))
        ping_one.start()
        ip_processes[ip_addr] = ping_one
    ip_list=[]
    for ip,process in ip_processes.items():
        if process.exitcode == 3:  # 退出码为3,说明PING成功
            ip_list.append(ip)
        else:
            process.terminate()
    return sorted(ip_list)  # 排序返回,因为有些先PING通
if __name__ == '__main__':
    import time
    t1 = time.time()
    active_ip = scapy_ping_scan('47.98.97.0/24')
    print('活动的IP地址为:')
    for ip in active_ip:
        print(ip)
    t2 = time.time()
    print(t2-t1)

一.动手写arp扫描

适合自连的扫描,不能扫描跨网段
arp只能扫自连,跨网络是不支持的,(二层)
arp.py

import logging

from scapy.layers.l2 import Ether, ARP

logging.getLogger("scapy.runtime").setLevel(logging.ERROR)
from scapy.all import *

def scapy_arp_request(ip_address, queue = None): #queue是多进程里面用的, ifname = 'eno'
   #dst原mark地址,出口不用填写,会自动填写上,ARP操作码为1,目的mark不知道和目的ip自己给的.二层接口(iface)多网卡一定要填,timeout设置最多等待时间,verbose设置自己打印,三层sr1,verbose一般设置false
    result_raw = srp(Ether(dst='FF:FF:FF:FF:FF')/ARP(op=1,hwdst='00:00:00:00:00', pdst=ip_address),timeout = 1, verbose =False)# ,iface = ifname
    print(result_raw)
    # result_raw是一个特殊的类(收到结果的包,和未收到结果的包)
    try:
        result_list = result_raw[0].res #收到结果的数据包,res把包做成一个list,0是收到的结果,1是未收到的结果
        if queue == None:
            return result_list[0][1].getlayer(ARP).fields['hwsrc'] # [s  r,s  r,s  r]
        else: # 有队列的情况
            queue.put((ip_address, result_list[0][1].getlayer(ARP).fields['hwsrc']))  # hwsrc表示源mark地址信息,向队列里面传ip和mark
    except:
        return
if __name__ == "__main__":
    import sys
    print(scapy_arp_request('192.168.13.134'))

arp1.py

import logging
logging.getLogger("scapy.runtime").setLevel(logging.ERROR)

import ipaddress
import time
from arp import scapy_arp_request
from multiprocessing import Process,Queue
def scapy_arp_scan(network):
    qyt_queue = Queue()
    net = ipaddress.ip_network(network)
    for ip in net:
        ip_addr = str(ip)
        arp_one = Process(target=scapy_arp_request,args=(ip_addr,qyt_queue))
        arp_one.start()
    time.sleep(2)
    ip_mac_list = []
    while True:
        if qyt_queue.empty():
            break
        else:
            ip,mac = qyt_queue.get()
            ip_mac_list.append((ip,mac))
    return ip_mac_list
if __name__==  '__main__':
    import sys
    active_ip_mac = scapy_arp_scan(sys.argv[1])
    print('活动的IP与mac地址如下:')
    for ip,mac in active_ip_mac:
        print(ip,mac)
